var productExceptSelf = function (nums) {
  var rear = [];
  var output = [];
  var len = nums.length;
  var mul = 1;
  for(var i = len - 1; i > 0; i--) { //从后往前累乘
    mul *= nums[i];
    rear[len - 1 - i] = mul;
  }
  mul = 1;
  for(var i = 0; i < len - 1; i++) {  //从前往后累乘
    mul *= nums[i];
    nums[i] = mul;
  }
  for(var i = 0; i < len; i++) {
    if(i == 0) {
      output[i] = rear[len - 2];
    }
    else if(i == len - 1) {
      output[i] = nums[len - 2];
    }
    else {
      output[i] = nums[i - 1] * rear[len - i - 2];
    }
  }
  return output;
};
